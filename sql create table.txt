CREATE TABLE PLAYERS
(
	GUID_PLAYER    UUID NOT NULL PRIMARY KEY
	,FIST_NAME     VARCHAR(100)
	,MID_NAME      VARCHAR(100)
	,LAST_NAME     VARCHAR(100)
    ,MAIL          VARCHAR(100)
	,PASSWORD_PL   bytea 
);

CREATE TABLE HEROES
(
	ID_HERO    		  INT  GENERATED ALWAYS AS IDENTITY PRIMARY KEY
   ,NAME_HERO         VARCHAR(50)
   ,RACE              VARCHAR(20)
   ,LEVEL_HERO        INT
   ,CLASS_HERO        INT
   ,SEX               VARCHAR(20)
   ,CURRENTEXP        INT
   ,HEIGHT            INT
   ,WEIGHT            INT
   ,SIZE              INT
   ,VISION            INT
   ,LANGUAGE_HERO     INT
   ,GUID_PLAYER       UUID NOT NULL references PLAYERS
);


   
CREATE TABLE HEROES_BASE_PARAM_DIC
(ID_FIELD  INT  GENERATED ALWAYS AS IDENTITY  PRIMARY KEY
,NAME_FIELD VARCHAR(20) NOT NULL);


CREATE TABLE HEROES_BASE_PARAM_ADDFL_INT
(
	ID_PLAYER     INT REFERENCES HEROES
   ,ID_FIELD      INT REFERENCES HEROES_BASE_PARAM_DIC
   ,FIELD         INT
   , PRIMARY KEY (ID_PLAYER ,ID_FIELD ));
